**Then add `publish-release` job** (port from corporate-build.yml, lines 264-321):
```yaml
publish-release:
  needs: [build-linux, build-windows, build-macos]
  if: github.ref_type == 'tag'
  runs-on: ubuntu-latest
  permissions:
    contents: write
  steps:
    - uses: actions/checkout@v4
    - name: Download all platform artifacts
      uses: actions/download-artifact@v4
      with:
        path: artifacts
    - name: List artifacts
      run: |
        ls -la artifacts/
        find artifacts/ -type f \( -name "*.dmg" -o -name "*.exe" -o -name "*.AppImage" -o -name "*.deb" -o -name "*.tar.gz" -o -name "*.zip" -o -name "latest-*.yml" -o -name "*.blockmap" \) | head -40
    - name: Create GitHub Release with artifacts
      env:
        GH_TOKEN: ${{ secrets.GITHUB_TOKEN }}
      run: |
        TAG_NAME="${GITHUB_REF_NAME}"
        mapfile -t FILES < <(find artifacts/ -type f \( -name "*.dmg" -o -name "*.exe" -o -name "*.AppImage" -o -name "*.deb" -o -name "*.tar.gz" -o -name "*.zip" -o -name "latest-*.yml" -o -name "*.blockmap" \))
        echo "Creating release ${TAG_NAME} with ${#FILES[@]} artifact(s)"

        if [ -f CHANGELOG.md ]; then
          NOTES_FROM_CHANGELOG=$(awk "/^## \[?${TAG_NAME#v}/{flag=1; next} /^## \[?[0-9]/{flag=0} flag" CHANGELOG.md)
        fi

        if [ -n "${NOTES_FROM_CHANGELOG:-}" ]; then
          echo "${NOTES_FROM_CHANGELOG}" > release-notes.md
        else
          git tag -l --format='%(contents)' "${TAG_NAME}" > release-notes.md
        fi

        printf '\n\n___\n%s\n%s\n%s\n%s\n%s\n%s\n' \
          "**Build posture:** corporate-minimal (default)" \
          "- OAuth providers: enabled (Google/Apple/Microsoft)" \
          "- Stripe billing UI: **disabled**" \
          "- Referral program UI: **disabled**" \
          "- Third-party realtime ASR: **disabled**" \
          "- Realtime via corporate backend (set OPENWHISPR_BACKEND_URL when building)" \
          >> release-notes.md

        gh release create "${TAG_NAME}" "${FILES[@]}" \
          --title "${TAG_NAME}" \
          --notes-file release-notes.md \
          --verify-tag
```

**Why these specific choices:**
- `--publish never` + dedicated publish-release job: aggregates all 4 builds (linux, win, mac-arm64, mac-x64) into ONE Release with ALL assets. With `--publish always` electron-builder races to create the release, often producing duplicates or partial uploads.
- Including `latest-*.yml` and `*.blockmap` in artifacts: required for electron-updater auto-update feed to work after release.
- `shell: bash` on Windows version-resolve step: pwsh (Windows runner default) chokes on `[[ ]]` test syntax.
- `--config.extraMetadata.version`: per CLAUDE.md, package.json `version` MUST stay at upstream baseline; tag is source of truth for Yambr version.

Use `Write` (not `Edit`) since this is a wholesale replacement.
