```js
// Phase 07 PLAN-04: tests for the Phase 05-02 empty-URL guard in
// openaiRealtimeStreaming.js. The guard prevents corporate builds from
// crashing on first record when streaming is enabled but no realtime URL
// resolved — instead, it throws a clear error pointing to the two recovery
// knobs (set the URL, or disable streaming).
//
// Mocking strategy: vi.doMock the generated build-config.cjs per test so we
// can swap empty/non-empty URL between cases. require.cache reset is needed
// because openaiRealtimeStreaming.js destructures the URL at module load.

const { vi, describe, test, expect, beforeEach, afterEach } = require("vitest");
const path = require("path");

const STREAMING_PATH = path.resolve(
  __dirname,
  "../../src/helpers/openaiRealtimeStreaming.js"
);
const CONFIG_PATH = path.resolve(
  __dirname,
  "../../src/config/build-config.generated.cjs"
);

function loadStreamingWithMockedUrl(url) {
  vi.resetModules();
  // Clear node's require cache for both modules so the destructure re-runs.
  delete require.cache[STREAMING_PATH];
  delete require.cache[CONFIG_PATH];
  vi.doMock(CONFIG_PATH, () => ({
    OPENWHISPR_REALTIME_WSS_URL: url,
    // Other keys touched by the module — keep minimal, default to empty/true
    // unless the module reads them at load time.
  }));
  // eslint-disable-next-line global-require
  return require(STREAMING_PATH);
}

afterEach(() => {
  vi.resetModules();
  vi.unmock(CONFIG_PATH);
  delete require.cache[STREAMING_PATH];
  delete require.cache[CONFIG_PATH];
});

describe("openaiRealtimeStreaming — empty-URL guard (Phase 05-02)", () => {
  test("connect() throws when OPENWHISPR_REALTIME_WSS_URL is empty", () => {
    const Streaming = loadStreamingWithMockedUrl("");
    // Minimal construction — adjust per Task 1 investigation findings.
    // If the export is the class directly: `new Streaming(opts)`.
    // If wrapped in {Class}: `new Streaming.OpenAIRealtimeStreaming(opts)`.
    const ctor = typeof Streaming === "function" ? Streaming : Streaming.OpenAIRealtimeStreaming || Streaming.default;
    const inst = new ctor({ apiKey: "sk-test-key" });
    expect(() => inst.connect()).toThrow(/OPENWHISPR_REALTIME_WSS_URL is empty/);
  });

  test("error message mentions both recovery knobs (URL var AND streaming-disable var)", () => {
    const Streaming = loadStreamingWithMockedUrl("");
    const ctor = typeof Streaming === "function" ? Streaming : Streaming.OpenAIRealtimeStreaming || Streaming.default;
    const inst = new ctor({ apiKey: "sk-test-key" });

    let err;
    try {
      inst.connect();
    } catch (e) {
      err = e;
    }
    expect(err).toBeDefined();
    // Recovery knob 1: how to provide a URL
    expect(err.message).toMatch(/OPENWHISPR_BACKEND_URL/);
    expect(err.message).toMatch(/OPENWHISPR_REALTIME_WSS_URL/);
    // Recovery knob 2: how to disable streaming entirely
    expect(err.message).toMatch(/OPENWHISPR_STREAMING=false/);
  });

  test("non-empty URL bypasses the empty-URL guard (different failure expected, e.g. WS connect)", () => {
    const Streaming = loadStreamingWithMockedUrl("wss://corp.example.com/v1/realtime");
    const ctor = typeof Streaming === "function" ? Streaming : Streaming.OpenAIRealtimeStreaming || Streaming.default;
    const inst = new ctor({ apiKey: "sk-test-key" });

    // We expect connect() to NOT throw the empty-URL message. It MAY throw
    // some other error (WebSocket connection failure, missing audio context,
    // etc.) — that's fine and out of scope for this guard test.
    let err;
    try {
      inst.connect();
    } catch (e) {
      err = e;
    }
    if (err) {
      expect(err.message).not.toMatch(/OPENWHISPR_REALTIME_WSS_URL is empty/);
    }
    // Best-effort cleanup if connect started any background work — ignore errors.
    try { inst.disconnect?.(); } catch {}
    try { inst.close?.(); } catch {}
  });
});
```
